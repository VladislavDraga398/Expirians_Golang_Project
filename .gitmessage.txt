# Типы коммитов (следуйте Conventional Commits)
# feat:     новая функциональность
# fix:      исправление ошибки
# docs:     изменения в документации
# style:    форматирование, стиль (без изменения логики)
# refactor: рефакторинг без изменения функционала
# perf:     улучшение производительности
# test:     добавление/правка тестов
# build:    сборка, зависимости, конфиги
# ci:       настройки CI/CD
# chore:    прочие изменения (без влияния на код приложения)
# revert:   откат предыдущего коммита
#
# Схема сообщения коммита:
# <type>(<scope>): <subject в повелительном наклонении>
#
# Примеры:
# feat(grpc): добавить RPC RefundOrder с валидацией
# fix(saga): исправить компенсацию при отмене подтверждённого заказа
# docs(readme): описать Makefile и запуск через docker-compose
# ci: добавить GitHub Actions (lint, test, build)
#
# Тело коммита (опционально): мотивация, детали реализации, ссылки на задачи
#
# BREAKING CHANGE: описание изменения API/контракта
